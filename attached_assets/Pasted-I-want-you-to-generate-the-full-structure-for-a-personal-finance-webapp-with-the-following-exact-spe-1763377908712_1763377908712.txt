DATABASE SCHEMA (implement inside schema.prisma)

Use SQLite.

model Account {
  id           Int      @id @default(autoincrement())
  name         String
  type         String    // cash, bank, credit_card, ewallet
  balance      Float
  creditLimit  Float?    // nullable
  transactions Transaction[]
  installments Installment[]
}

model Transaction {
  id            Int      @id @default(autoincrement())
  date          DateTime
  description   String
  amount        Float
  category      String
  type          String    // fixed, variable, installment, goal, income
  account       Account   @relation(fields: [accountId], references: [id])
  accountId     Int
  installment   Installment? @relation(fields: [installmentId], references: [id])
  installmentId Int?
}

model Installment {
  id           Int     @id @default(autoincrement())
  name         String
  startDate    DateTime
  term         Int
  monthlyAmount Float
  monthsPaid   Int
  status       String   // active, done
  account      Account  @relation(fields: [accountId], references: [id])
  accountId    Int
  transactions Transaction[]
}

model Goal {
  id            Int      @id @default(autoincrement())
  name          String
  targetAmount  Float
  currentAmount Float
  deadline      DateTime
}
